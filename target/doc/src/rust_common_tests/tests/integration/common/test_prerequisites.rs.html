1/// Check test prerequisites
2/// # Errors
3/// Returns error if basic project structure is missing
4pub fn check_test_prerequisites() -> Result<(), String> {
5    let CURRENT_DIR =
6        std::env::current_dir().map_err(|e| format!("Failed to get current directory: {e}"))?;
7
8    // Skip if we're in the common tests directory itself
9    if CURRENT_DIR.file_name().unwrap_or_default() == "rust-common-tests" {
10        return Ok(());
11    }
12
13    // Check for basic project structure
14    let SRC_DIR = CURRENT_DIR.join("src");
15    if !SRC_DIR.exists() {
16        return Err("src directory not found".to_string());
17    }
18
19    let CARGO_PATH = CURRENT_DIR.join("Cargo.toml");
20    if !CARGO_PATH.exists() {
21        return Err("Cargo.toml not found".to_string());
22    }
23
24    Ok(())
25}

27/// Check if service config exists
28/// # Panics
29/// Panics if current directory cannot be determined
30#[must_use]
31#[allow(dead_code)]
32pub fn has_service_config() -> bool {

34    let SERVICE_CONFIG_PATH = CURRENT_DIR.join("service.toml");
35    SERVICE_CONFIG_PATH.exists()
36}
